%-------------------- CUSTOM CHAPTER FORMATTING ----------------------
% Customize chapter title format to be ALL CAPS and centered
\titleformat{\chapter}[block]
  {\normalfont\bfseries\LARGE\centering}       % Format for whole title
  {\MakeUppercase{\chaptername}\ \thechapter:} % Label (CHAPTER 1:)
  {0.5em}                                      % Separation between label and title
  {\MakeUppercase}                             % Before-code -> makes title ALL CAPS
% \titleformat{command}[shape]{format}{label}{sep}{before-code}[after-code]
% - command: the sectioning command to be formatted (e.g., \chapter)

% Customize chapter title spacing
\titlespacing{\chapter}{0pt}{0pt}{24pt}
% Parameters: command, left margin, space above chapter title, space below chapter title
% - 0pt: no left margin indentation
% - 0pt: vertical space above the chapter title
% - 24pt: vertical space below the chapter title
% ----------------------------------------------------------------------------------------------

% -------------------- CUSTOM SECTION FORMATTING ----------------------
% Format section titles to be ALL CAPS
\titleformat{\section}[block]
  {\normalfont\bfseries}
  {\thesection}{0.5cm}{\MakeUppercase}

% Customize section title spacing
\titlespacing{\section}{0pt}{6pt}{0pt}
% ----------------------------------------------------------------------------------------------

% ------------------ CUSTOM SUBSECTION FORMATTING --------------------
% Format subsection titles to be bold (not ALL CAPS)
\titleformat{\subsection}[block]    % Run-in style (runin) can also be used
  {\normalfont\bfseries}
  {\thesubsection}{0.5cm}{}
  % runin: title continues on same line as number (no line break)

% Customize subsection title spacing
\titlespacing{\subsection}{0pt}{0pt}{-6pt}
% ----------------------------------------------------------------------------------------------

% ------------------------- BULLET LISTS FORMATTING ------------------------
% Adjust default spacing for all bullet lists (itemize)
\setlist[itemize]{
    nosep,      % removes default top/bottom space (prevents conflicts)
    topsep=\ENUMITEMTOPSEP, % Space above/below the bullet list
    itemsep=\ITEMSEP\ % Vertical space between individual bullet points
}

% Adjust default spacing for all bullet lists (enumerate)
\setlist[enumerate]{
    nosep,      % removes default top/bottom space (prevents conflicts)
    topsep=\ENUMITEMTOPSEP, % Space above/below the bullet list
    itemsep=\ITEMSEP\ % Vertical space between individual bullet points
}
